rule all:
	input:
		["fastqFiles/{ech}.1_1.fastq".format(ech=ech) for ech in ECHANTILLONS],
		["fastqFiles/{ech}.1_2.fastq".format(ech=ech) for ech in ECHANTILLONS]

	priority: 50	
	output: 
		[""sraFiles/{ech}.1"/{ech}.1".format(ech=ech) for ech in ECHANTILLONS]
	shell:
		"""
		list="SRR628582 SRR628583 SRR628584 SRR628585 SRR628586 SRR628587 SRR628588 SRR628589"; for i in $list; do wget -P sraFiles/ https://sra-downloadb.be-md.ncbi.nlm.nih.gov/sos1/sra-pub-run-5/$i/$i.1; done
		"""

	input:
		"sraFiles/{sample}.1"
	output:
		"fastqFiles/{sample}.1_1.fastq",
		"fastqFiles/{sample}.1_2.fastq"
	singularity:

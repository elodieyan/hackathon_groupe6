SAMPLES =["SRR628582","SRR628583","SRR628584","SRR628585","SRR628586","SRR628587","SRR628588","SRR628589"]

rule all:
	input:
		expand("results/fastqFiles/{sample}.1_1.fastq", sample = SAMPLES),
		expand("results/fastqFiles/{sample}.1_2.fastq", sample = SAMPLES)

rule makefastq:
	input:
		expand("results/sraFiles/{sample}.1", sample = SAMPLES)
	output:
		"results/fastqFiles/{sample}.1_1.fastq",
		"results/fastqFiles/{sample}.1_2.fastq"
	singularity: 
		"docker://evolbioinfo/sratoolkit:v2.10.8"
	shell: 
		"""
		fastq-dump --gzip --split-files {input} > {output}		
		"""		

rule download:
	output: 
		"results/sraFiles/{sample}.1"
	shell:
		"""
		wget -O {output} https://sra-downloadb.be-md.ncbi.nlm.nih.gov/sos1/sra-pub-run-5/{wildcards.sample}/{wildcards.sample}.1
		"""
